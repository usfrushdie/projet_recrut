CREATE TABLE PERSONNES (EMAIL VARCHAR(50) PRIMARY KEY, 
						MOT_DE_PASSE VARCHAR(50), 
						NOM VARCHAR(50), 
						PRENOM VARCHAR(50), 
						STATUT CHAR(1) CHECK(STATUT IN ('C', 'R')));


CREATE TABLE INFO_COMP (EMAIL VARCHAR(50) PRIMARY KEY,
						POSTE_ACTUEL VARCHAR(50), 
						ADRESSE VARCHAR(50), 
						TEL VARCHAR(50), 
						FOREIGN KEY (EMAIL) REFERENCES PERSONNES(EMAIL));
						
CREATE TABLE EXPERIENCES  ( EMAIL VARCHAR(50) PRIMARY KEY,
							TITRE VARCHAR(50), 
							ENTREPRISE VARCHAR(50), 
							VILLE VARCHAR(50), 
						    DATE_DEBUT DATE,
						    DATE_FIN DATE,
							DESCRIPTION TEXT,
							FOREIGN KEY (EMAIL) REFERENCES PERSONNES(EMAIL));

CREATE TABLE FORMATIONS   ( EMAIL VARCHAR(50) PRIMARY KEY,
							ETABLISSEMENT VARCHAR(50), 
							DIPLOME VARCHAR(50), 
							VILLE VARCHAR(50), 
						    DATE_DEBUT DATE,
						    DATE_FIN DATE,
							DESCRIPTION TEXT,
							FOREIGN KEY (EMAIL) REFERENCES PERSONNES(EMAIL));

CREATE TABLE PROJETS   	  ( EMAIL VARCHAR(50) PRIMARY KEY,
							NOM_PROJET VARCHAR(50), 
							DESCRIPTION TEXT,
							FOREIGN KEY (EMAIL) REFERENCES PERSONNES(EMAIL));

CREATE TABLE COMPETENCES  ( EMAIL VARCHAR(50) PRIMARY KEY,
							TITRE VARCHAR(50), 
							FOREIGN KEY (EMAIL) REFERENCES PERSONNES(EMAIL));

CREATE TABLE OFFRES 	  ( ID_OFFRE SERIAL PRIMARY KEY,
							EMAIL VARCHAR(50),
							TITRE VARCHAR(50),
						    DESCRIPTION TEXT,
							FOREIGN KEY (EMAIL) REFERENCES PERSONNES(EMAIL));

CREATE TABLE CANDIDATURES ( ID_OFFRE INTEGER,
							EMAIL VARCHAR(50),
							FOREIGN KEY (EMAIL) REFERENCES PERSONNES(EMAIL), 
						   	FOREIGN KEY (ID_OFFRE) REFERENCES OFFRES(ID_OFFRE),
						  	PRIMARY KEY (EMAIL, ID_OFFRE));
